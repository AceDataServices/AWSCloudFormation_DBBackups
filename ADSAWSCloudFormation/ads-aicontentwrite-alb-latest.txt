  VpcId:
    Type: AWS::EC2::VPC::Id
    Default: vpc-5f27ea25
    Description: VPC ID for the load balancer
  
  SubnetIds:

  SecurityGroupId:

    Default: arn:aws:acm:us-east-1:481330377004:certificate/d779f372-f06b-4bcc-bcf8-8a1610a1b460

  Environment:

  ApplicationLoadBalancer:
    Type: AWS::ElasticLoadBalancingV2::LoadBalancer
    Properties:
      Name: ads-aicontentwrite-alb
      Type: application
      Scheme: internet-facing
      IpAddressType: ipv4
      Subnets: !Ref SubnetIds
      SecurityGroups:
        - !Ref SecurityGroupId
      LoadBalancerAttributes:

          Value: '180'
        - Key: routing.http2.enabled

        - Key: access_logs.s3.prefix
          Value: 'elb'

          Value: 'true'

        - Key: routing.http.xff_header_processing.mode
          Value: 'append'

          Value: '3600'

      HealthCheckPath: /

      HealthCheckProtocol: HTTP
      HealthCheckTimeoutSeconds: 5

        HttpCode: '200'
      Tags:

  TestTargetGroup:
    Type: AWS::ElasticLoadBalancingV2::TargetGroup

  # HTTP Listener (Port 80) - Redirects to HTTPS
  HTTPListener:
    Type: AWS::ElasticLoadBalancingV2::Listener
    Properties:
      LoadBalancerArn: !Ref ApplicationLoadBalancer
      Port: 80
      Protocol: HTTP
      DefaultActions:
        - Type: redirect
          RedirectConfig:
            Protocol: HTTPS
            Port: '443'
            StatusCode: HTTP_301

  HTTPSListener:

      Port: 443

      Certificates:

      DefaultActions:
        - Type: forward
          TargetGroupArn: !Ref MainTargetGroup

  # HTTPS Listener (Port 8088) - Secondary listener
  HTTPSListener8088:
